sil_score = []
cluster_list = np.arange(2,13)
for i in cluster_list:
    clusterer = KMeans(n_clusters=i)
    preds = clusterer.fit_predict((df[features_cluster]))
    score = silhouette_score(df[features_cluster], preds)
    sil_score.append(score)
